// Agent identifiers — one per diagnostic agent
type DiagnosticAgentId = "web_performance" | "seo" | "tech_stack" | "company_context";

  startedAt?: string;  // ISO 8601
  completedAt?: string;
  error?: string;      // Only set on failure — NOT exposed differently from "no data"
}

// Overall session status
type SessionStatus = "pending" | "running" | "completed" | "failed";

// Web Performance results (from PSI + CrUX APIs)
interface WebPerformanceResult {
  // Core Web Vitals
  lcp?: { value: number; unit: "ms"; rating: "good" | "needs-improvement" | "poor" };
  inp?: { value: number; unit: "ms"; rating: "good" | "needs-improvement" | "poor" };
  cls?: { value: number; unit: "unitless"; rating: "good" | "needs-improvement" | "poor" };
  // Performance scores (0-100)
  mobileScore?: number;
  desktopScore?: number;
  // CrUX data
  cruxData?: Record<string, unknown> | null;
  // Image audit findings
  imageAudit?: {
    issues: Array<{ type: string; description: string; severity: "high" | "medium" | "low" }>;
    totalImages: number;
  };
  // HTML size analysis
  htmlSize?: {
    totalBytes: number;
    frameworkPayloadBytes?: number;
    structuredDataBytes?: number;
  };
  // Cache analysis
  cacheHeaders?: Record<string, string>;
}

// SEO results (from HTML crawl)

  title?: string;
  metaDescription?: string;
  ogTags?: Record<string, string>;
  canonicalUrl?: string;
  headingStructure?: { tag: string; text: string }[];
  robotsMeta?: string;
  hasRobotsTxt?: boolean;
  hasSitemap?: boolean;
  structuredData?: unknown[];

**Create `apps/mesh/src/diagnostic/ssrf-validator.ts`** — SSRF prevention module. This is a security-critical BLOCKER.

**`normalizeUrl(input: string): { url: URL; normalized: string }`**
- Prepend `https://` if no protocol provided
- Reject non-HTTP(S) protocols (ftp://, file://, data:, javascript:, etc.) — throw with clear message
- Lowercase the hostname
- Remove default ports (:80 for http, :443 for https)
- Remove trailing slash from path
- Remove hash fragments
- Return both the parsed URL object and a normalized string for caching

**`validateUrl(input: string): Promise<{ url: URL; normalized: string }>`**
- Call `normalizeUrl` first
- Resolve hostname to IP addresses using `dns.promises.resolve4` and `dns.promises.resolve6` (try both, catch ENODATA/ENOTFOUND gracefully)
- Check ALL resolved IPs against private/reserved ranges:
  - `127.0.0.0/8` (loopback)
  - `10.0.0.0/8` (private)
  - `172.16.0.0/12` (private)
  - `192.168.0.0/16` (private)
  - `169.254.0.0/16` (link-local)
  - `0.0.0.0/8` (unspecified)
  - `::1` (IPv6 loopback)
  - `fc00::/7` (IPv6 private)
  - `fe80::/10` (IPv6 link-local)
- If ANY resolved IP is private → reject with error: `"URL resolves to a private/internal IP address"`
- If hostname is literally `localhost` → reject before DNS resolution
- If DNS resolution fails completely (no A or AAAA records) → reject with: `"Could not resolve hostname"`
- Return the validated URL and normalized string

**Helper: `isPrivateIp(ip: string): boolean`**
- Parse IPv4 octets and check ranges
- Parse IPv6 and check prefixes
- Handle IPv4-mapped IPv6 (::ffff:x.x.x.x)
